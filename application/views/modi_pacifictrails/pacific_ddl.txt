CREATE TABLE ACTIVITIES(
	ACTIVITYID VARCHAR(10) NOT NULL,
	ACTITVITYDESCRIPTION VARCHAR(100) NOT NULL,
	PRIMARY KEY (ACTIVITYID)
);

CREATE TABLE PACKAGE(
	PACKAGEID int NOT NULL AUTO_INCREMENT,
	PACKAGENAME varchar(30)
	PACKAGEDESCRIPTION VARCHAR(200) NOT NULL,
	OFNIGHTS INT,
	COST INT,
	PRIMARY KEY (PACKAGEID)	
);

CREATE TABLE RESERVATIONYURT(
	RESID int NOT NULL AUTO_INCREMENT,
	ARRIVALDATE DATE,
	OFNIGHTS INT,
	PACKAGEID int,
	PRIMARY KEY (RESID),
	FOREIGN KEY (PACKAGEID) REFERENCES PACKAGE(PACKAGEID)
);

CREATE TABLE client(
	clientid int NOT NULL AUTO_INCREMENT,
	FNAME VARCHAR(20) NOT NULL,
	LNAME VARCHAR(20) NOT NULL,
	ADDRESS VARCHAR(30),
	PHONE VARCHAR(10) NOT NULL,
	EMAIL VARCHAR(320) NOT NULL,
	RESID  INT,
	ACTIVITYID VARCHAR(10),
	PRIMARY KEY (CLIENTID),
	FOREIGN KEY (RESID) REFERENCES reservationyurt(RESID),
	FOREIGN KEY (ACTIVITYID) REFERENCES ACTIVITIES(ACTIVITYID)
);



CREATE TABLE WHENDATE(
	ID INT NOT NULL AUTO_INCREMENT,
	DATE DATE,
	ACTIVITYID VARCHAR(10),
	CLIENTID INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (ACTIVITYID) REFERENCES ACTIVITIES(ACTIVITYID),
	FOREIGN KEY (CLIENTID) REFERENCES CLIENT(CLIENTID)
);
